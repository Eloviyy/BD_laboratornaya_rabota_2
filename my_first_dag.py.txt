from datetime import datetime, timedelta
from airflow import DAG
from airflow.operators.python import PythonOperator

default_args = {
        'owner': 'student',
        'depends_on_past': False,
        'start_date': datetime(2024, 1, 1),
        'email_on_failure':  False,
        'email_on_retry': False,
        'retries': 1,
        'retry_delay': timedelta(minutes=5)
}
def say_hello():
        print("Здравствуйте, Владимир  Александрович!")

def say_world():
        print("Es mi primero DAG en Airflow!")

def calculate_sum():
        result = 4 + 6
        print("Результат вычисления 4 + 6 = {result}")

with DAG(
        dag_id='my_first_data_pipeline',
        default_args=default_args,
        description='Простой DAG для выполнения лабораторной работы №2',
        schedule='@daily',
        catchup=False,
        tags=['lab','tutorial']
) as dag:

        task1 = PythonOperator(task_id='print_hello', python_callable=say_hello)
        task2 = PythonOperator(task_id='print_world', python_callable=say_world)
        task3 = PythonOperator(task_id='calculate_numbers', python_callable=calculate_sum)

        task1 >> [task2, task3]
